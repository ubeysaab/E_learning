--create database E_LearningDb
use E_LearningDb

create table tbl_Instructors
(
 instructor_id int  primary key not null identity(1,1),
 instructor_name varchar(30),
 instructor_username varchar(50)
)
create table tbl_Students
(
 student_id int identity(1,1) not null primary key,
 student_name varchar(30),
 student_username varchar(50)

)

create table tbl_Courses
(
 course_id int primary key not null identity(1,1),
 instructor_id int,
 course_title varchar(350),
 course_content varchar(max)
  
)
create table tbl_Questions
(
 question_id int primary key  not null identity(1,1) ,
 course_id int,
 question varchar(450)
)
create table tbl_Answers
(
answer_id int primary key not null identity(1,1),
question_id int ,
correct_ans varchar(max),
option_a varchar(max),
option_b varchar(max),
option_c varchar(max)

)
create table tbl_Grades 
(
 grade_id int not null  primary key identity(1,1),
 course_id int ,
 student_id int, 
 question_id  int ,
 student_answer varchar(max),
 correct_answer varchar(max)
)
